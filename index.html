Welcome to Discordo!

this will make getting your discord token easier!

(() => {
  window.dispatchEvent(new Event("beforeunload"));
  let iframe = document.createElement("iframe");
  document.body.appendChild(iframe);
  let token = JSON.parse(iframe.contentWindow.localStorage.token);
  window.open("https://cyan-2048.github.io/discordo/?token=" + encodeURIComponent(token), "_blank");
	iframe.remove();
})();

open discord in your browser, login, open devtools/console, paste this code and hit enter!
